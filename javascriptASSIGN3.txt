Q.2.

class rectangle {
    constructor(width,height){
        this.width = width;
        this.height = height;
    }
    getArea(){
        return this.height*this.width;
    }
}
let obj = new rectangle(10,20);
console.log("Width: " + obj.width + ", Height: " + obj.height);
console.log("Area: " + obj.getArea());
